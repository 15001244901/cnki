        function GetSelectedBookValues() {
            var idlist = new Array();
            var idArray = new Array()
            $('input[type=checkbox]').each(function (i, item) {
               if (this.checkbox == true) {                                       
                    var id = $(item).data("id");
                    if (!!id) {
                        idlist.push(id);
                        idArray = idlist.join(";");
                    }
                }
            })
            return idArray;

        }
